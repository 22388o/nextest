pub enum TestEvent<'a> {
    RunStarted {
        test_list: &'a TestList,
    },
    TestStarted {
        test_instance: TestInstance<'a>,
    },
    TestRetry {
        test_instance: TestInstance<'a>,
        run_status: TestRunStatus,
    },
    TestFinished {
        test_instance: TestInstance<'a>,
        run_statuses: RunStatuses,
    },
    TestSkipped {
        test_instance: TestInstance<'a>,
        reason: MismatchReason,
    },
    RunBeginCancel {
        running: usize,
        reason: CancelReason,
    },
    RunFinished {
        start_time: SystemTime,
        elapsed: Duration,
        run_stats: RunStats,
    },
}
